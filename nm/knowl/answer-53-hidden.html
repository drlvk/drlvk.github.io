f = @(t, y) -t*y;   % or without -
h = 0.3;   %  or 0.1, 0.01 
t = 0:h:3;
y0 = 1;
y = y0*ones(size(t));

for k = 1:numel(t)-1
    pred = y(k) + h*f(t(k), y(k));
    y(k+1) = y(k) + h/2*(f(t(k), y(k)) + f(t(k+1), pred));
end

exact = exp(-t.^2 /2);   % or without -
plot(t, y, t, exact)
legend('trapezoid', 'exact')
